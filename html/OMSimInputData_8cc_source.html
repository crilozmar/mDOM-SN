143 G4double OMSimInputData::Spice_Absorption(G4double pLambd){
144  G4double lKappa = 1.08;
145  G4double lParamA = 6954./m;
146  G4double lParamB = 6618*nm;
147  G4double lAdust = 1./(mSpice_a400inv[mSpiceDepth_pos])*pow(pLambd/(400.*nm), - lKappa);
148  G4double lDeltaTau = Spice_Temperature(mSpice_Depth[mSpiceDepth_pos])-Spice_Temperature(1730.);
149  G4double la_inv = 1./(lAdust+lParamA*exp(-lParamB/pLambd)*(1.+0.01*lDeltaTau));
150  return la_inv;
151 }

158 G4double OMSimInputData::Spice_Refraction(G4double pLambd){
